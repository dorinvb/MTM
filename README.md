function buildHome(){
  const grid=document.getElementById('ch-grid');

  let totalKnown=0;

  Object.entries(CAT_META).forEach(([cat,meta])=>{

  document.getElementById('stat-known').textContent=totalKnown;
}

let fQs=[],fIdx=0,fKnown=new Set(),fCat='',fFlipped=false;

function startFlash(cat){
  fCat=cat;
  const cp=getCP(cat);
  fKnown=new Set(cp.known||[]);
  const all=byCat(cat);
  // unknown first (shuffled), then known (shuffled)
  const unk=all.filter(q=>!fKnown.has(q.id)).sort(()=>Math.random()-.5);
  const kn=all.filter(q=>fKnown.has(q.id)).sort(()=>Math.random()-.5);
  fQs=[...unk,...kn];
  fIdx=0;
  document.getElementById('flash-ch-name').textContent=cat;
  renderFlash();showPage('flash');
}

function renderFlash(){
  if(!fQs.length) return;
  const q=fQs[fIdx];
  document.getElementById('flash-q').textContent=q.question;
  document.getElementById('flash-a-intro').textContent=q.answer_intro||'';
  const ul=document.getElementById('flash-a-list');
  ul.innerHTML=(q.answer_bullets||[]).map(b=>`<li>${b}</li>`).join('');
  ul.style.display=q.answer_bullets&&q.answer_bullets.length?'flex':'none';

  // reset flip
  fFlipped=false;
  document.getElementById('card-3d').classList.remove('flipped');
  document.getElementById('know-row').style.display='none';

  const pct=((fIdx+1)/fQs.length*100).toFixed(1);
  document.getElementById('flash-prog-bar').style.width=pct+'%';
  document.getElementById('flash-counter').textContent=`${fIdx+1} / ${fQs.length}`;
  document.getElementById('flash-prog-txt').textContent=`${fIdx+1}/${fQs.length}`;
  document.getElementById('flash-prev').disabled=fIdx===0;
  document.getElementById('flash-next').disabled=fIdx===fQs.length-1;
  document.getElementById('flash-hint').style.display='none';
  updateFStats();
}
